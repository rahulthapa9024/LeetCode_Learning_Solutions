//Increasing Triplet Subsequence

//Medium

//My approch leaving some test cases

class Solution {
public:
    bool increasingTriplet(vector<int>& nums) {
        for(int i =0;i<nums.size();i++){
            if(nums[i] < nums[i+1]){
                if(nums[i+1] < nums[i+2]){
                    return true;
                }
            }
        }

        return false;
    }
};




// optimal Solution O(n)
class Solution {
public:
    bool increasingTriplet(vector<int>& nums) {
        int first  = INT_MAX;
        int second = INT_MAX;

        for(int num : nums){
            if(num <= first){
                first = num;
            }
            else if(num <= second){
                second = num;
            }
            else{
                return true;
            }
        }
        return false;
    }
};
