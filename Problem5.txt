Can Place Flowers
Easy

here we can plant more than given plants so for that condition also return true 

class Solution {
public:
    bool canPlaceFlowers(vector<int>& flowerbed, int n) {
        int count = 0;
        int size = flowerbed.size();

        for(int i = 0; i < size; i++){
            if(flowerbed[i] == 0 && size ==1){  // if the size is 1 and no plant is planted
                count++;
                flowerbed[i] = 1;
            }

            if(i == 0 && flowerbed[i] == 0 && flowerbed[i+1] == 0){ // first position has no plant and next is no plant as well
                count++;
                flowerbed[i] = 1;
            }

            if(i > 0 && i < size -1 && flowerbed[i] == 0 && flowerbed[i+1] == 0 && flowerbed[i-1] == 0){ // for empty plot in the middle 
                 count++;
                flowerbed[i] = 1;
            }

            if(i == size -1 && flowerbed[i] == 0 && flowerbed[i-1] == 0){ // last position has no plant and prev is no plant as well
                count++;
                flowerbed[i] = 1;
            }

        }
        return count >= n; //can plant more than given plants so for that condition also return true 
    }
};
