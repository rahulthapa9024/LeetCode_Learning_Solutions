Reverse Words in a String
Medium


//My approch 
class Solution {
public:
    string reverseWords(string s) {
        vector<string> temp;
        string result = "";
        int len = s.length();
        int i = 0;

        while(i < len) {
            string t = "";

            // skip spaces
            while(i < len && s[i] == ' ') {
                i++;
            }

            int j = i;

            // collect word
            while(j < len && s[j] != ' ') {
                t += s[j];
                j++;
            }

            if(t != "") {
                temp.push_back(t);
            }

            i = j;
        }

        // reverse order
        for(int k = temp.size() - 1; k >= 0; k--) {
            result += temp[k];

            if(k != 0) {
                result += " ";
            }
        }

        return result;
    }
};

        }

        for(int i = temp.size() - 1 ; i >= 0; i--){ // reverse iterating the temp and storeing words in reverse in result

            }
        }
