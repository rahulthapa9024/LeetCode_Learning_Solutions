//Kids With the Greatest Number of Candies

//My Solution

class Solution {
public:
    vector<bool> kidsWithCandies(vector<int>& candies, int extraCandies) {
       vector<bool> ans;
       int n = candies.size();

       for(int i=0; i<n ;i++){

            int temp = candies[i] + extraCandies;
            int max = temp;

        for(int j = 0 ; j<n; j++){

            if(temp < candies[j]){
                max = candies[j];
            }

        }
        if(max == temp){
            ans.push_back(true);
        }
        else{
             ans.push_back(false);
        }
       }
       return ans;

    }
};

//optimal Solution

class Solution {
public:
    vector<bool> kidsWithCandies(vector<int>& candies, int extraCandies) {
        vector<bool> result;
        int maxVal = 0;

        for (int c : candies){
            maxVal = max(maxVal,c);
        }

        for(int c : candies){
            result.push_back(c + extraCandies >= maxVal );
        }
    return result;
    }
};
